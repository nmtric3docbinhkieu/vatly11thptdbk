Vấn đề React state không update sau khi fetch data từ Supabase
🔍 Chẩn đoán vấn đề
Lỗi setLeaderboard is not defined cho thấy vấn đề nằm ở scope của biến trong component. State leaderboard được khai báo nhưng function setter không được truyền đúng cách.

🎯 Root Cause
Trong index.html, bạn đang tạo state:

javascript
const [leaderboard, setLeaderboard] = useState([]);
Nhưng khi sử dụng trong phần render, bạn đang gọi setLeaderboard từ một scope không có quyền truy cập đến setter này.

✅ Giải pháp triệt để
1. Sửa lại cấu trúc component
Trong index.html, cập nhật phần xử lý end game:

// Debug React state
window.debugReactState = function() {
    // Tìm React root instance
    const rootElement = document.getElementById('root');
    const rootKey = Object.keys(rootElement).find(key => key.startsWith('__reactContainer$'));
    
    if (!rootKey) {
        console.error('Không tìm thấy React instance');
        return;
    }
    
    const root = rootElement[rootKey];
    let fiber = root;
    
    // Tìm fiber node của App component
    while (fiber && (!fiber.elementType || fiber.elementType.name !== 'App')) {
        fiber = fiber.child;
    }
    
    if (fiber && fiber.memoizedState) {
        console.log('=== REACT STATE DEBUG ===');
        
        // Duyệt qua các hooks
        let hook = fiber.memoizedState;
        let hookIndex = 0;
        
        while (hook) {
            const hookName = getHookName(hookIndex);
            console.log(`Hook #${hookIndex} (${hookName}):`, hook.memoizedState);
            hook = hook.next;
            hookIndex++;
        }
        
        console.log('=== END DEBUG ===');
    } else {
        console.error('Không tìm thấy App component state');
    }
};

// Helper để đoán tên hook dựa trên index (ước lượng)
function getHookName(index) {
    const hookNames = [
        'gameState',           // 0
        'currentIdx',          // 1
        'score',               // 2
        'selected',            // 3
        'showExpl',            // 4
        'questions',           // 5
        'selectedChapter',     // 6
        'chapter3Activity',    // 7
        'selectedExperiment',  // 8
        'chapter4Activity',    // 9
        'selectedChapter4Experiment', // 10
        'cheatWarnings',       // 11
        'showCheatAlert',      // 12
        'isTestActive',        // 13
        'startTime',           // 14
        'elapsedTime',         // 15
        'leaderboard',         // 16
        'showLeaderboard',     // 17
        'loadingLeaderboard',  // 18
        'student',             // 19
        'adminMode',           // 20
        'adminPassword',       // 21
        'adminError',          // 22
        'authMode',            // 23
        'authForm',            // 24
        'authError',           // 25
        'authLoading',         // 26
        'registerSuccess',     // 27
        'exportLoading'        // 28
    ];
    
    return hookNames[index] || `unknown_${index}`;
}

// Force update component
window.forceUpdateApp = function() {
    const rootElement = document.getElementById('root');
    const rootKey = Object.keys(rootElement).find(key => key.startsWith('__reactContainer$'));
    
    if (!rootKey) {
        console.error('Không tìm thấy React container');
        return;
    }
    
    const root = rootElement[rootKey];
    
    // Tìm và force update App component
    function findAndUpdate(fiber) {
        if (!fiber) return;
        
        if (fiber.elementType && fiber.elementType.name === 'App') {
            console.log('Tìm thấy App component, forcing update...');
            
            // Force update bằng cách gọi setState
            if (fiber.memoizedState) {
                // Tìm hook leaderboard (index 16)
                let hook = fiber.memoizedState;
                for (let i = 0; i < 16 && hook; i++) {
                    hook = hook.next;
                }
                
                if (hook) {
                    const currentLeaderboard = hook.memoizedState;
                    console.log('Current leaderboard:', currentLeaderboard);
                    
                    // Trigger update bằng cách set lại giá trị
                    const setLeaderboard = hook.queue.dispatch;
                    if (setLeaderboard) {
                        setLeaderboard([...currentLeaderboard]); // Clone array để trigger update
                        console.log('Force update triggered!');
                    }
                }
            }
        }
        
        findAndUpdate(fiber.child);
        findAndUpdate(fiber.sibling);
    }
    
    findAndUpdate(root);
    console.log('Force update complete!');
};

📊 Best Practices để tránh lỗi tương tự
1. Tách component hợp lý
Không nên để quá nhiều logic trong một component

Tạo các component con riêng cho từng màn hình

2. Sử dụng React DevTools
Cài extension React DevTools để debug state

Kiểm tra component hierarchy và props

3. Kiểm tra scope của setter
javascript
// ✅ Đúng - setter được khai báo trong cùng scope
const [leaderboard, setLeaderboard] = useState([]);
const fetchData = async () => {
    const data = await getData();
    setLeaderboard(data); // ✅ Có thể truy cập
}

// ❌ Sai - setter không có trong scope
const SomeComponent = () => {
    const [leaderboard, setLeaderboard] = useState([]);
    
    return <ChildComponent onComplete={(data) => {
        setLeaderboard(data); // ❌ Không thể truy cập từ bên ngoài
    }} />
}
4. Logging có cấu trúc
javascript
// Thêm logging function
window.logState = function(name, value) {
    console.log(`[State Update] ${name}:`, value);
    console.trace(); // Xem call stack
};

// Sử dụng
setLeaderboard(data);
window.logState('leaderboard', data);
5. Checklist khi gặp lỗi state không update
Setter có được định nghĩa trong cùng scope không?

Component có bị unmount trước khi setState không?

Có đang set state trong async function không đúng cách?

Có đang mutate state trực tiếp không?

Component có nhận đúng props không?

🚀 Kết luận
Vấn đề chính là do cấu trúc component không tối ưu và scope của setter. Giải pháp triệt để:

Tách component - Mỗi màn hình là một component riêng

Props drilling - Truyền setter và data xuống component con

Debug tools - Thêm công cụ debug để kiểm tra state real-time

Logging - Ghi log mỗi khi state thay đổi

Sau khi áp dụng các thay đổi trên, React sẽ re-render đúng cách mỗi khi state leaderboard được cập nhật.
